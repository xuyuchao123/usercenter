select * from (
        select QRCODE_STR as qRCodeStr, GMT_CREATE as gmtCreate
        from SYS_QRCODE_JL
        <where>
            <if test="qrCodeStr != null">
                and QRCODE_STR = #{qrCodeStr}
            </if>
        </where>
          order by GMT_CREATE) a
        where a.rownum > 1
